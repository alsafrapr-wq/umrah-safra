function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("Data"); var p = e.parameter; // فحص التكرار var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (data[i][1] == p.id_number) { return ContentService.createTextOutput(JSON.stringify({"result": "duplicate"})).setMimeType(ContentService.MimeType.JSON); } } // رفع الصورة لـ Google Drive var imageUrl = "لا يوجد"; if (p.id_photo && p.id_photo !== "") { try { var splitData = p.id_photo.split(","); var contentType = p.id_photo_type; var base64Data = splitData[1]; var decodedData = Utilities.base64Decode(base64Data); var blob = Utilities.newBlob(decodedData, contentType, "ID_" + p.id_number); var file = DriveApp.createFile(blob); imageUrl = file.getUrl(); } catch(err) { imageUrl = "خطأ في الرفع: " + err.toString(); } } // إضافة الصف sheet.appendRow([ new Date(), p.id_number, p.name, p.phone, p.birth_date, p.gender, p.nationality, imageUrl ]); return ContentService.createTextOutput(JSON.stringify({"result": "success"})).setMimeType(ContentService.MimeType.JSON); } function doGet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var settingsSheet = ss.getSheetByName("Settings"); var settings = settingsSheet.getRange("B1:B3").getValues(); return ContentService.createTextOutput(JSON.stringify({ "registration_status": settings[0][0], "female_status": settings[1][0], "image_required": settings[2][0] })).setMimeType(ContentService.MimeType.JSON); }
